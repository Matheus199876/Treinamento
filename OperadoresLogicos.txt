# Aritméticos
a = 10
b = 3

print(a + b)  # Soma
print(a - b)  # Subtração
print(a * b)  # Multiplicação
print(a / b)  # Divisão
print(a // b) # Divisão inteira
print(a % b)  # Resto da divisão
print(a ** b) # Potência

# Comparação
print(a == b)  # Igual?
print(a != b)  # Diferente?
print(a > b)   # Maior?
print(a < b)   # Menor?
